/*
COMENTARIOS = */ /* ou * ;

Criação da tabela vendacarros
DATA = criar tabela vendacarros;
INPUT = criação de colunas;
CARD = inserir informações;

$ = colocar um $ depois do nome para identificar que é uma string, e não Numero.

*/
-----------------------------------------------------------------------------
DATA vendacarros;
INPUT CARRO$ ANO JAN FEV MAR;

CIVIC 2018 123 89 90

GOL 2011 89 115 168
SAVEIRO 2016 98 56 153

RUN;
